	Let the subset $E\subset \mathbb{R}$ defined by
	$\displaystyle E=\Bigg\{2\left(1+\frac 1 n\right)
	^{\hspace{-1mm}n}: n\in \mathbb{N}\setminus\{0\}\Bigg\}$.
	\\Then
	\begin{choices}
		\correctchoice{$10$ is a majorant of $E$}
		\wrongchoice{the minimum of $E$ is $2$}
		\wrongchoice{$E$ is closed}
		\wrongchoice{the supremum of $E$ belongs to $E$}
	\end{choices}
\end{question}
