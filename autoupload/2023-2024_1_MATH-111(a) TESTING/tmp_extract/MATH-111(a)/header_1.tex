{\large \textbf{Notation} }\smallskip
%*******************************************************************%

\noindent-- Pour une matrice $A$, $a^{}_{ij}$ désigne \lelement\ situé sur la ligne $i$ et la colonne $j$ de~la~matrice.

\noindent-- Pour un vecteur $\vec x\in\RR^n$, $x^{}_i$ désigne la $i$-ème \composante de $\vec x$.

\noindent-- $\Id{m}$ désigne la matrice identité \detaille$m{\times}m$.

\noindent-- $\Poly{n}$ désigne l'espace vectoriel des polynômes réels de degré inférieur ou égal à $n$.

\noindent-- $\MatricesSpace{m}{n}{\RR}$ désigne l'espace vectoriel des matrices \detaille$m{\times}n$\acoeffreels.%point.

\ifx\ScalNotation\undefined\else{\noindent-- Pour $\vec{x},\vec{y}\in\RR^n$, le \pseuclidien\ est défini par $\ScalProd{\vec{x}}{\vec{y}}=x^{}_1y^{}_1 + \ldots+x^{}_ny^{}_n$.}\fi

\ifx\NormNotation\undefined\else{\noindent-- Pour $\vec{x}\in\RR^n$, la norme euclidienne est définie par $\left\|\vec{x}\right\| = \sqrt{\ScalProd{\vec{x}}{\vec{x}}\,\,}$.}\fi

{\large \textbf{Notation} (all standard) }\smallskip
%*******************************************************************%

\noindent-- $\RR $ denotes the set of real numbers.

\noindent-- For a matrix $A$, $a^{}_{ij}\in\RR$ denotes the entry of $A$ in row $i$ and column $j$.

\noindent-- For a vector $\vec x\in\RR^n$, $x^{}_i$ denotes the $i$th coordinate of $\vec x$.

\noindent-- $\Id{m}$ denotes the $m{\times}m$ identity matrix.

\noindent-- $\Poly{n}$ is the vector space of polynomials of degree less than or equal to $n$.

\noindent-- $\MatricesSpace{m}{n}{\RR}$ is the vector space of~$m{\times}n$ matrices.
